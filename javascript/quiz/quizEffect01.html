$("#quizConfirm").on("click", function() {
    const userAnswer = $("#quizInput").val();
    const infoAnswer = ""; // 여기에 infoAnswer 변수의 값이 들어가야 합니다.    
    if (infoAnswer === userAnswer) {
      $("#quizAnimation").addClass("like");
    } else {
      $("#quizAnimation").addClass("dislike");
    }   
    $("#quizAnswer").removeClass("none");
    $("#quizDesc").removeClass("none");
    $("#quizConfirm").addClass("none");
    $("#quizInput").addClass("none");
});
